# Welch Periodogram

D = 50
L = 100
fs = [fs1, fs2]
signals = [hrm, x]
Pper = []

for i in range(2):
    frame_samples = L*fs[i]
    frames = int((len(signals[i])/frame_samples)*L/D)
    pper = []
    
    # Bartlett window
    wb = np.bartlett(frame_samples)
    
    for j in range(frames-1):
        start = j * int(frame_samples/2)
        finish = start + frame_samples
        
        pper.append(periodogram(wb * signals[i][start:finish]))

    Pper.append(sum(pper)/frames)
